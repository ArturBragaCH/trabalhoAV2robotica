#include 

LiquidCrystal lcd(2, 3, 4, 5, 6, 7);
String estadoPilar1, estadoPilar2;;
int ledVermelho = 9, ledVerde = 8;
int pilar1, pilar2;

void setup(){
  lcd.begin(16,2);
  pinMode(A5, INPUT);
  pinMode(A4, INPUT);
  pinMode(ledVerde, OUTPUT);
  pinMode(ledVermelho, OUTPUT);
  pinMode(10, OUTPUT);
}

void mostrarEstado(){
	lcd.clear();
    lcd.setCursor(0, 0);
  	lcd.print("Pilar 1: "+ estadoPilar1);
    lcd.setCursor(0, 1);
  	lcd.print("Pilar 2: "+ estadoPilar2);
}

String verificarEstado(int pilar){
  if (pilar < 5){
	return "Ok"; 
  } else if (pilar <= 10){
 	return "Irregular"; 
  } else {
    return "Comprometido";
  }
}

void loop(){
  pilar1 = map(analogRead(A5), 585, 372, 0, 90);
  pilar2 = map(analogRead(A4), 585, 372, 0, 90);
  
  estadoPilar1 = verificarEstado(pilar1);
  estadoPilar2 = verificarEstado(pilar2);
  
  
  if (estadoPilar1 == "Ok" & estadoPilar2 == "Ok"){
    digitalWrite(ledVermelho, LOW);
    digitalWrite(ledVerde, HIGH);
    noTone(10);
	mostrarEstado();
  } else {
    digitalWrite(ledVermelho, HIGH);
    digitalWrite(ledVerde, LOW);
    tone(10, 4450, 1000);
	mostrarEstado();
  }
  
  delay(100);
}
