Sobre o Projeto

As estruturas com maior possibilidade de apresentar risco de queda incluem marquises, lajes, estruturas em balanço, pontes, telhados, coberturas e outras construções que estão constantemente sujeitas a esforços e à ação do tempo. O risco de colapso dessas estruturas pode ser causado por diversos fatores, como falhas no planejamento e dimensionamento estrutural, utilização de materiais inadequados ou de baixa qualidade, desgaste natural dos componentes ao longo dos anos, infiltrações, corrosão das armaduras, fissuras, trincas e excesso de carga aplicada sobre a estrutura. Além disso, a falta de inspeções periódicas e de manutenção preventiva contribui significativamente para o surgimento de problemas que podem comprometer a estabilidade da construção. Em muitos casos, os acidentes envolvendo quedas de estruturas estão relacionados à negligência humana, seja na fase de projeto, execução, utilização ou conservação da edificação.

Montagem e Esquema

Descrição passo a passo:

  1. Coloque o Arduino Uno na área de trabalho.
  2. Conecte o LED ao pino digital 13 com resistor de 220Ω ao GND.
  3. Verifique conexões e carregue o código.

Código Arduino

#include 

LiquidCrystal lcd(2, 3, 4, 5, 6, 7);
String estadoPilar1, estadoPilar2;;
int ledVermelho = 9, ledVerde = 8;
int pilar1, pilar2;

void setup(){
  lcd.begin(16,2);
  pinMode(A5, INPUT);
  pinMode(A4, INPUT);
  pinMode(ledVerde, OUTPUT);
  pinMode(ledVermelho, OUTPUT);
  pinMode(10, OUTPUT);
}

void mostrarEstado(){
	lcd.clear();
    lcd.setCursor(0, 0);
  	lcd.print("Pilar 1: "+ estadoPilar1);
    lcd.setCursor(0, 1);
  	lcd.print("Pilar 2: "+ estadoPilar2);
}

String verificarEstado(int pilar){
  if (pilar < 5){
	return "Ok"; 
  } else if (pilar <= 10){
 	return "Irregular"; 
  } else {
    return "Comprometido";
  }
}

void loop(){
  pilar1 = map(analogRead(A5), 585, 372, 0, 90);
  pilar2 = map(analogRead(A4), 585, 372, 0, 90);
  
  estadoPilar1 = verificarEstado(pilar1);
  estadoPilar2 = verificarEstado(pilar2);
  
  
  if (estadoPilar1 == "Ok" & estadoPilar2 == "Ok"){
    digitalWrite(ledVermelho, LOW);
    digitalWrite(ledVerde, HIGH);
    noTone(10);
	mostrarEstado();
  } else {
    digitalWrite(ledVermelho, HIGH);
    digitalWrite(ledVerde, LOW);
    tone(10, 4450, 1000);
	mostrarEstado();
  }
  
  delay(100);
}
